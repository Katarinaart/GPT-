`````python
import re

with open('index_full_content.html', 'r', encoding='utf-8') as f:
    html = f.read()

errors = []

# TEST 1: All sections exist

    if f'<section class="section" id="section{i}">' not in html:
        errors.append(f"Section {i} missing")

    pos = html.find(f'<section class="section" id="section{i}">')

    errors.append("Sections out of order")

    errors.append(f"Info boxes: {html.count('class=\"info-box\"')} (expected ≥30)")
`````

`````javascript
// Quiz functionality
function checkQuiz(quizId) {
  const quiz = document.getElementById(`quiz-${quizId}`);
  const questions = quiz.querySelectorAll('.question');
  let correct = 0;
  let total = questions.length;
  
  questions.forEach((question, index) => {
    const correctAnswer = question.dataset.correct;
    const selected = quiz.querySelector(`input[name="q${quizId}-${index+1}"]:checked`);
    const answers = question.querySelectorAll('.answer');
    
    // Remove previous styling
    answers.forEach(ans => {
      ans.classList.remove('correct', 'incorrect');
    });
    
    if (selected) {
      const selectedAnswer = selected.value;
      const selectedLabel = selected.closest('.answer');
      
      if (selectedAnswer === correctAnswer) {
        correct++;
        selectedLabel.classList.add('correct');
      } else {
        selectedLabel.classList.add('incorrect');
        // Highlight correct answer
        answers.forEach(ans => {
          if (ans.querySelector('input').value === correctAnswer) {
            ans.classList.add('correct');
          }
        });
      }
    } else {
      // No answer selected - show correct
      answers.forEach(ans => {
        if (ans.querySelector('input').value === correctAnswer) {
          ans.classList.add('correct');
        }
      });
    }
  });
  
  // Show result
  const resultDiv = document.getElementById(`result-${quizId}`);
  const percentage = (correct / total) * 100;
  
  resultDiv.classList.remove('excellent', 'good', 'poor');
  resultDiv.classList.add('show');
  
  if (percentage >= 80) {
    resultDiv.classList.add('excellent');
    resultDiv.innerHTML = `🎉 Отлично! ${correct}/${total} (${percentage.toFixed(0)}%)`;
  } else if (percentage >= 60) {
    resultDiv.classList.add('good');
    resultDiv.innerHTML = `👍 Хорошо! ${correct}/${total} (${percentage.toFixed(0)}%)`;
  } else {
    resultDiv.classList.add('poor');
    resultDiv.innerHTML = `📚 Нужно повторить. ${correct}/${total} (${percentage.toFixed(0)}%)`;
  }
  
  // Scroll to result
  resultDiv.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
}
`````

`````python
def convert_test_to_html(test_markdown, test_id):
    questions = parse_questions(test_markdown)
    html = f'<div class="quiz" id="quiz-{test_id}">\n'
    html += f'  <h2 class="quiz-title">📝 Тест {test_id}: {title}</h2>\n\n'
    
    for i, q in enumerate(questions, 1):
        correct_letter = q.correct_answer  # 'a', 'b', 'c', 'd'
        html += f'  <div class="question" data-correct="{correct_letter}">\n'
        html += f'    <h3>Вопрос {i}</h3>\n'
        html += f'    <p>{q.text}</p>\n'
        html += f'    <div class="answers">\n'
        
        for letter, answer in zip(['a','b','c','d'], q.answers):

            html += f'        <input type="radio" name="q{test_id}-{i}" value="{letter}">\n'
`````
